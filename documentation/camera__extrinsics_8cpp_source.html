00125 // Convert a world frame point into the camera frame.
00126 void CameraExtrinsics::WorldToCamera(double wx, double wy, double wz,
00127                                      double* cx, double* cy, double* cz) const {
00128   if (cx == nullptr || cy == nullptr || cz == nullptr) return;
00129 
00130   Eigen::Vector4d w_h = Eigen::Vector4d();
00131   w_h << wx, wy, wz, 1.0;
00132 
00133   const Eigen::Vector4d c_h = CameraExtrinsics::WorldToCamera().Project(w_h);
00134 
00135   *cx = c_h(0);
00136   *cy = c_h(1);
00137   *cz = c_h(2);
00138 }
