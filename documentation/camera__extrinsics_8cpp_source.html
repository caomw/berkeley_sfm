125 // Convert a world frame point into the camera frame.
126 void CameraExtrinsics::WorldToCamera(double wx, double wy, double wz,
127  double* cx, double* cy, double* cz) const {
128  if (cx == nullptr || cy == nullptr || cz == nullptr) return;
129 
130  Eigen::Vector4d w_h = Eigen::Vector4d();
131  w_h << wx, wy, wz, 1.0;
132 
133  const Eigen::Vector4d c_h = CameraExtrinsics::WorldToCamera().Project(w_h);
134 
135  *cx = c_h(0);
136  *cy = c_h(1);
137  *cz = c_h(2);
138 }

143  if (wx == nullptr || wy == nullptr || wz == nullptr) return;

149 
150  *wx = w_h(0);
