00273 bool CameraIntrinsics::CameraToImage(double cx, double cy, double cz,
00274                                      double *u_distorted,
00275                                      double *v_distorted) const {
00276   if (u_distorted == nullptr || v_distorted == nullptr) return false;
00277 
00278   // We can't project points that lie behind the camera.

00289 bool CameraIntrinsics::DirectionToImage(double u_normalized,
00290                                         double v_normalized,
00291                                         double *u_distorted,
00292                                         double *v_distorted) const {
00293   if (u_distorted == nullptr || v_distorted == nullptr) return false;
00294 
00295   // Distort the normalized direction vector;
00296   double u = 0.0, v = 0.0;
00297   Distort(u_normalized, v_normalized, &u, &v);
00298 
00299   // Flip v - image space coordinates begin at the top.
00300   v = -v;
00301 
00302   // Make a homogeneous vector from the output.
00303   Eigen::Vector3d p = Eigen::Vector3d();
00304   p << u, v, 1.0;
00305 
00306   // Multiply the distorted direction vector by camera intrinsic matrix to get
00307   // the image space point.
00308   const Eigen::Vector3d p_out = CameraIntrinsics::IntrinsicsMatrix() * p;
00309   *u_distorted = p_out(0);
00310   *v_distorted = p_out(1);
00311 
00312   // Make sure that the resulting point is in the image.
00313   return PointInImage(*u_distorted, *v_distorted);
00314 }

00316 void CameraIntrinsics::ImageToDirection(double u_distorted, double v_distorted,
00317                                         double *u_normalized,
00318                                         double *v_normalized) const {
00319   if (u_normalized == nullptr | v_normalized == nullptr) return;
00320 
00321   // Make a homogeneous image space point.
00322   Eigen::Vector3d p_distorted = Eigen::Vector3d();
00323   p_distorted << u_distorted, v_distorted, 1.0;
00324 
00325   // Multiply the distorted homogeneous image space point by the inverse
00326   // of the camera intrinsic matrix to get a distorted ray.
00327   const Eigen::Vector3d p = CameraIntrinsics::InverseIntrinsicsMatrix() * p_distorted;
00328 
00329   // Undistort the ray to get the normalized direction vector.
00330   Undistort(p(0), p(1), u_normalized, v_normalized);
00331 }

00333 // Warp a point into the image.
00334 void CameraIntrinsics::Distort(double u, double v, double *u_distorted,
00335                                double *v_distorted) const {
00336   if (u_distorted == nullptr | v_distorted == nullptr) return;
00337 
00338   // Get the camera's radial distortion (see OpenCV help page at the top of
00339   // this file).
00340   const double r_sq = u * u + v * v;
00341   const double radial_dist =
00342       1.0 + k1_ * r_sq + k2_ * r_sq * r_sq + k5_ * r_sq * r_sq * r_sq;
00343 
00344   // If radial distortion is too extreme, our 5th-order model might not be a
00345   // good fit. Instead, estimate the distortion with a simpler model (linear),
00346   // and apply radial corrections.
00347   if (radial_dist < 0.85 || radial_dist > 1.15) {
00348     // Distortion too extreme. Warp the point with estimated radial
00349     // distortion.
00350     const double radius = hypot(image_width_, image_height_);
00351     *u_distorted = u / sqrt(r_sq) * radius;
00352     *v_distorted = v / sqrt(r_sq) * radius;
00353   } else {
00354     // Compute radial distortion with 5th order model.
00355     const double dx0 = 2.0 * k3_ * u * v + k4_ * (r_sq + 2.0 * u * u);
00356     const double dx1 = k3_ * (r_sq + 2.0 * v * v) + 2.0 * k4_ * u * v;
00357 
00358     // Homogeneous distorted direction vector.
00359     *u_distorted = radial_dist * u + dx0;
00360     *v_distorted = radial_dist * v + dx1;
00361   }
00362 }

00364 // Rectilinearize point.
00365 void CameraIntrinsics::Undistort(double u_distorted, double v_distorted,
00366                                  double *u, double *v, int iterations) const {
00367   if (u == nullptr || v == nullptr) return;
00368 
00369   // Iteratively attempt to undo the radial distortion (see OpenCV help page
00370   // at the top of this file).
00371   double u_refine = u_distorted;
00372   double v_refine = v_distorted;
00373   for (int i = 0; i < iterations; ++i) {
00374     const double r_sq = u_refine * u_refine + v_refine * v_refine;
00375     const double dx0 = 2.0 * k3_ * u_refine * v_refine +
00376                        k4_ * (r_sq + 2.0 * u_refine * u_refine);
00377     const double dx1 = k3_ * (r_sq + 2.0 * v_refine * v_refine) +
00378                        2.0 * k4_ * u_refine * v_refine;
00379     u_refine = u_distorted - dx0;
00380     v_refine = v_distorted - dx1;
00381   }
00382 
00383   // Return the undistorted direction vector.
00384   *u = u_refine;
00385   *v = v_refine;
00386 }
