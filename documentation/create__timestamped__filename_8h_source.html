00030 // Creates a filename from base_name and extension that includes the current
00031 // date and time. The date and time format is specified by format_string, with a
00032 // default value of %Y%m%d_%H%M%s.
00033 // e.g.
00034 // base_name = filename
00035 // extension = .ext
00036 // format_string = %Y%m%d-%H%M%S
00037 // output: filename_20150702_094750.ext
00038 //
00039 // If the system clock is not able to give us a time, default to
00040 // basename.extension.
00041 inline std::string CreateTimestampedFilename(const std::string &base_name,
00042                                              const std::string &extension,
00043                                              const char *format_string =
00044                                                  "%Y%m%d-%H%M%S") {
00045   std::string output_filename, date_time_string;
00046 
00047   // Get the current date.
00048   int size = 12;
00049   char *date_time = static_cast<char *>(malloc(size));
00050   date_time[0] = '\0';
00051   time_t now = time(NULL);
00052 
00053   // If time exists, format date string.
00054   if (now != -1) {
00055     // Make sure there is enough room in the buffer to get the full date and
00056     // time (specified by format_string). If there is not enough room, retry
00057     // until there is.
00058     int success = strftime(date_time, size - 1, format_string, gmtime(&now));
00059     while (!success) {
00060       free(date_time);
00061       size += 10;
00062       date_time = static_cast<char *>(malloc(size));
00063       date_time[0] = '\0';
00064       success = strftime(date_time, size - 1, format_string, gmtime(&now));
00065     }
00066     date_time_string = Join("_", std::string(date_time));
00067   }
00068   free(date_time);
00069   return Join(base_name, date_time_string, ".", extension);
00070 }
