00053 // When extracting N-dimensional descriptors from a set of M keypoints, OpenCV
00054 // will store the descriptors in a (M-K)xN matrix, where K is the number of
00055 // keypoints that OpenCV failed to compute a descriptor for. In other words,
00056 // rows correspond to keypoints, and columns to indices of the descriptor.

00084 template <typename Descriptor>
00085 bool DescriptorExtractor<Descriptor>::SetDescriptor(
00086     const std::string& descriptor_type) {
00087   // Set the descriptor type.
00088   descriptor_type_ = descriptor_type;
00089   bool valid_descriptor_type = true;
00090 
00091   // Create an OpenCV descriptor extractor.
00092   if (descriptor_type.compare("SIFT") == 0) {
00093     extractor_ = cv::DescriptorExtractor::create("SIFT");
00094   } else if (descriptor_type_.compare("SURF") == 0) {
00095     extractor_ = cv::DescriptorExtractor::create("SURF");
00096   } else if (descriptor_type_.compare("BRIEF") == 0) {
00097     extractor_ = cv::DescriptorExtractor::create("BRIEF");
00098   } else if (descriptor_type_.compare("BRISK") == 0) {
00099     extractor_ = cv::DescriptorExtractor::create("BRISK");
00100   } else if (descriptor_type_.compare("FREAK") == 0) {
00101     extractor_ = cv::DescriptorExtractor::create("FREAK");
00102   } else {
00103     VLOG(1) << "Descriptor type \"" << descriptor_type
00104             << "\"is not available. Defaulting to SIFT.";
00105     extractor_ = cv::DescriptorExtractor::create("SIFT");
00106     valid_descriptor_type = false;
00107     descriptor_type_ = "SIFT";
00108   }
00109 
00110   return valid_descriptor_type;
00111 }

00113 template <typename Descriptor>
00114 bool DescriptorExtractor<Descriptor>::DescribeFeatures(
00115     const Image& image, KeypointList& keypoints,
00116     std::vector<Feature>& features_out,
00117     std::vector<Descriptor>& descriptors_out) {
00118 
00119   // Make the user has called SetDescriptor().
00120   if (descriptor_type_.empty()) {
00121     VLOG(1)
00122         << "Descriptor has not been specified via SetDescriptor(). Failed to "
00123            "extract descriptors.";
00124     return false;
00125   }
00126 
00127   CHECK(extractor_) << "The descriptor extractor is null.";
00128   features_out.clear();
00129 
00130   // Convert the input image to OpenCV's format. Note that descriptors must be
00131   // extracted on the grayscale image, and that the image format must be CV_8U.
00132   cv::Mat cv_image;
00133   image.ToCV(cv_image);
00134   cv_image.convertTo(cv_image, CV_8U, 255);
00135 
00136   // Extract descriptors from the provided keypoints in the image.
00137   cv::Mat cv_descriptors;
00138   try {
00139     extractor_->compute(cv_image, keypoints, cv_descriptors);
00140   } catch (const std::exception& e) {
00141     VLOG(1) << "Failed to extract descriptors: " << e.what();
00142     return false;
00143   }
00144 
00145   // Convert the computed OpenCV-type descriptors and keypoints into a list of
00146   // features and descriptors.
00147   for (size_t ii = 0; ii < keypoints.size(); ++ii) {
00148     Feature feature;
00149     feature.u_ = keypoints[ii].pt.x;
00150     feature.v_ = keypoints[ii].pt.y;
00151     features_out.push_back(feature);
00152 
00153     Descriptor descriptor;
00154     OpenCVToEigenVec<Descriptor>(cv_descriptors.row(ii), descriptor);
00155     descriptors_out.push_back(descriptor);
00156   }
00157 
00158   return true;
00159 }
