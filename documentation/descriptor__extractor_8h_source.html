84 template <typename Descriptor>
85 bool DescriptorExtractor<Descriptor>::SetDescriptor(
86  const std::string& descriptor_type) {
87  // Set the descriptor type.
88  descriptor_type_ = descriptor_type;
89  bool valid_descriptor_type = true;
90 
91  // Create an OpenCV descriptor extractor.
92  if (descriptor_type.compare("SIFT") == 0) {
93  extractor_ = cv::DescriptorExtractor::create("SIFT");
94  } else if (descriptor_type_.compare("SURF") == 0) {
95  extractor_ = cv::DescriptorExtractor::create("SURF");
96  } else if (descriptor_type_.compare("BRIEF") == 0) {
97  extractor_ = cv::DescriptorExtractor::create("BRIEF");
98  } else if (descriptor_type_.compare("BRISK") == 0) {
99  extractor_ = cv::DescriptorExtractor::create("BRISK");
100  } else if (descriptor_type_.compare("FREAK") == 0) {
101  extractor_ = cv::DescriptorExtractor::create("FREAK");
102  } else {
103  VLOG(1) << "Descriptor type \"" << descriptor_type
104  << "\"is not available. Defaulting to SIFT.";
105  extractor_ = cv::DescriptorExtractor::create("SIFT");
106  valid_descriptor_type = false;
107  descriptor_type_ = "SIFT";
108  }
109 
110  return valid_descriptor_type;
111 }

113 template <typename Descriptor>
114 bool DescriptorExtractor<Descriptor>::DescribeFeatures(
115  const Image& image, KeypointList& keypoints,
116  std::vector<Feature>& features_out,
117  std::vector<Descriptor>& descriptors_out) {
118 
119  // Make the user has called SetDescriptor().
120  if (descriptor_type_.empty()) {
121  VLOG(1)
122  << "Descriptor has not been specified via SetDescriptor(). Failed to "
123  "extract descriptors.";
124  return false;
125  }
126 
127  CHECK(extractor_) << "The descriptor extractor is null.";
128  features_out.clear();
129 
130  // Convert the input image to OpenCV's format. Note that descriptors must be
131  // extracted on the grayscale image, and that the image format must be CV_8U.
132  cv::Mat cv_image;
133  image.ToCV(cv_image);
134  cv_image.convertTo(cv_image, CV_8U, 255);
135 
136  // Extract descriptors from the provided keypoints in the image.
137  cv::Mat cv_descriptors;
138  try {
139  extractor_->compute(cv_image, keypoints, cv_descriptors);
140  } catch (const std::exception& e) {
141  VLOG(1) << "Failed to extract descriptors: " << e.what();
142  return false;
143  }
144 
145  // Convert the computed OpenCV-type descriptors and keypoints into a list of
146  // features and descriptors.
147  for (size_t ii = 0; ii < keypoints.size(); ++ii) {
148  Feature feature;
149  feature.u_ = keypoints[ii].pt.x;
150  feature.v_ = keypoints[ii].pt.y;
151  features_out.push_back(feature);
152 
153  Descriptor descriptor;
154  OpenCVToEigenVec<Descriptor>(cv_descriptors.row(ii), descriptor);
155  descriptors_out.push_back(descriptor);
156  }
157 
158  return true;
159 }
