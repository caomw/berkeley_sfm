60 // Compute the L2 norm of the difference between two descriptor vectors. If both
61 // descriptors have unit length, the L2 norm is equal to 2*(1-x.y). Since all
62 // distances are computed this way, we can drop the leading 2*. The L2 norm
63 // induces an inner product space over R^{n}, and we can test as such.
64 struct ScaledL2Distance {
65  // Descriptor type that the ScaledL2Distance functor operates on.
66  typedef Eigen::VectorXf Descriptor;
67 
68  // Functor method.
69  float operator()(const Descriptor& descriptor1,
70  const Descriptor& descriptor2) {
71  CHECK_EQ(descriptor1.size(), descriptor2.size());
72  return 1.0f - descriptor1.dot(descriptor2);
73  }
74 
75  // Descriptors need normalization.
76  static bool MaybeNormalizeDescriptors(std::vector<Descriptor>& descriptors) {
77  for (auto& descriptor : descriptors) {
78  descriptor.normalize();
79  }
80  return true;
81  }
82 }; //\struct ScaledL2Distance

84 // Compute the Hamming distance between two binary descriptor vectors. This is
85 // the number of bits that are in disagreement (i.e. bit1 ^ bit2 == 1).
86 struct HammingDistance {
87  // Descriptor type that the HammingDistance functor operates on.
88  typedef Eigen::Matrix<unsigned char, Eigen::Dynamic, 1> Descriptor;
89 
90  // Functor method.
91  int operator()(const Descriptor& descriptor1, const Descriptor& descriptor2) {
92  CHECK_EQ(descriptor1.size(), descriptor2.size());
93  int sum = 0;
94  for (size_t ii = 0; ii < descriptor1.size(); ++ii)
95  sum += descriptor1(ii) ^ descriptor2(ii);
96  return sum;
97  }
98 
99  // Descriptors don't need normalization.
100  static bool MaybeNormalizeDescriptors(std::vector<Descriptor>& descriptors) {
101  return false;
102  }
103 }; //\struct HammingDistance
