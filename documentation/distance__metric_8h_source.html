00060 // Compute the L2 norm of the difference between two descriptor vectors. If both
00061 // descriptors have unit length, the L2 norm is equal to 2*(1-x.y). Since all
00062 // distances are computed this way, we can drop the leading 2*. The L2 norm
00063 // induces an inner product space over R^{n}, and we can test as such.
00064 struct ScaledL2Distance {
00065   // Descriptor type that ScaledL2Distance functor operates on.
00066   typedef Eigen::VectorXf Descriptor;
00067 
00068   // Functor method.
00069   float operator()(const Descriptor& descriptor1,
00070                     const Descriptor& descriptor2) {
00071     CHECK_EQ(descriptor1.size(), descriptor2.size());
00072     return 1.0f - descriptor1.dot(descriptor2);
00073   }
00074 
00075   static bool RequiresNormalizedDescriptors() { return true; }
00076 };  //\struct ScaledL2Distance
