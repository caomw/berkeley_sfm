60 // Compute the L2 norm of the difference between two descriptor vectors. If both
61 // descriptors have unit length, the L2 norm is equal to 2*(1-x.y). Since all
62 // distances are computed this way, we can drop the leading 2*. The L2 norm
63 // induces an inner product space over R^{n}, and we can test as such.
64 struct ScaledL2Distance {
65  // Descriptor type that ScaledL2Distance functor operates on.
66  typedef Eigen::VectorXf Descriptor;
67 
68  // Functor method.
69  float operator()(const Descriptor& descriptor1,
70  const Descriptor& descriptor2) {
71  CHECK_EQ(descriptor1.size(), descriptor2.size());
72  return 1.0f - descriptor1.dot(descriptor2);
73  }
74 
75  static bool RequiresNormalizedDescriptors() { return true; }
76 }; //\struct ScaledL2Distance
