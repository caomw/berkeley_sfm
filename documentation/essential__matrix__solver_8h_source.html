00082 // Compute the essential matrix from a fundamental matrix and camera intrinsics.
00083 Eigen::Matrix3d EssentialMatrixSolver::ComputeEssentialMatrix(Eigen::Matrix3d F,
00084                                   CameraIntrinsics cam1,
00085                                   CameraIntrinsics cam2) {
00086   // Extract intrinsics matrices.
00087   Eigen::Matrix3d K1, K2;
00088   K1 = cam1.IntrinsicsMatrix();
00089   K2 = cam2.IntrinsicsMatrix();
00090 
00091   // Calculate the essential matrix.
00092   Eigen::Matrix3d E = K2.transpose() * F * K1;
00093   return E;
00094 }
