00040 // This class defines a base class for feature matching. A derived class should
00041 // be used to implement the specific feature matching strategy. The feature
00042 // matcher is templated by a distance metric that will be used to compare
00043 // distances between descriptor vectors. All matching strategies will attempt to
00044 // do pairwise matches between all pairs of images. This is slow, with O(n^2) in
00045 // the number of images.
00046 //

00066 // Template by the type of distance metric that will be used to compare
00067 // descriptors.
00068 template <typename DistanceMetric>
00069 class FeatureMatcher {
00070  public:
00071   // Each DistanceMetric operates on a specific type of descriptor vector.
00072   typedef typename DistanceMetric::Descriptor Descriptor;
00073 
00074   FeatureMatcher() { }
00075   virtual ~FeatureMatcher() { }
00076 
00077   // Add a single image's features for matching.
00078   virtual inline void AddImageFeatures(
00079       const std::vector<Feature>& image_features,
00080       const std::vector<Descriptor>& image_descriptors);
00081 
00082   // Add features from a set of images for matching.
00083   virtual inline void AddImageFeatures(
00084       const std::vector<std::vector<Feature> >& image_features,
00085       const std::vector<std::vector<Descriptor> >& image_descriptors);
00086 
00087   // Match images together using the input options.
00088   virtual inline bool MatchImages(const FeatureMatcherOptions& options,
00089                                   PairwiseImageMatchList& image_matches);
00090 
00091  protected:
00092   // Abstract method to match a pair of images using the input options. Override
00093   // this in the derived feature matching strategy class to implement it.
00094   virtual bool MatchImagePair(int image_index1, int image_index2,
00095                               std::vector<FeatureMatch>& feature_matches) = 0;
00096 
00097   // Find the set intersection of the two sets of input feature matches, and
00098   // store that set in the second argument.
00099   virtual inline void SymmetricMatches(
00100       const std::vector<LightFeatureMatch>& feature_matches_lhs,
00101       std::vector<LightFeatureMatch>& feature_matches_rhs);
00102 
00103   // Each image has a list of features and descriptors (one per feature).
00104   std::vector<std::vector<Feature> > image_features_;
00105   std::vector<std::vector<Descriptor> > image_descriptors_;
00106 
00107   // A set of options used for matching features and images.
00108   FeatureMatcherOptions options_;
00109 
00110  private:
00111   DISALLOW_COPY_AND_ASSIGN(FeatureMatcher)
00112 };  //\class FeatureMatcher

00139 template <typename DistanceMetric>
00140 bool FeatureMatcher<DistanceMetric>::
00141 MatchImages(const FeatureMatcherOptions& options,
00142             PairwiseImageMatchList& image_matches) {
00143   // Store the matching options locally.
00144   options_ = options;
00145 
00146   // Iterate over all pairs of images and attempt to match them together by
00147   // comparing their features.
00148   for (size_t ii = 0; ii < image_features_.size(); ++ii) {
00149     // Make sure this image has features.
00150     if (image_features_[ii].size() == 0) {
00151       continue;
00152     }
00153     for (size_t jj = ii+1; jj < image_features_.size(); ++jj) {
00154       // Make sure this image has features.
00155       if (image_features_[jj].size() == 0) {
00156         continue;
00157       }
00158 
00159       // Create an image match object and attempt to match image ii to image jj.
00160       PairwiseImageMatch image_match;
00161       if (!MatchImagePair(ii, jj, image_match.feature_matches_)) {
00162         VLOG(1) << "Could not match image " << ii << " to image " << jj << ".";
00163         continue;
00164       }
00165 
00166       // If the image match was successful, store it.
00167       image_match.image1_index_ = ii;
00168       image_match.image2_index_ = jj;
00169       image_matches.push_back(image_match);
00170     }
00171   }
00172 
00173   // Return whether or not we found matches between any of the images.
00174   return image_matches.size() > 0;
00175 }

00177 template <typename DistanceMetric>
00178 void FeatureMatcher<DistanceMetric>::SymmetricMatches(
00179     const std::vector<LightFeatureMatch>& feature_matches_lhs,
00180     std::vector<LightFeatureMatch>& feature_matches_rhs) {
00181 
00182   std::unordered_map<int, int> feature_indices;
00183   feature_indices.reserve(feature_matches_lhs.size());
00184 
00185   // Add all LHS matches to the map.
00186   for (const auto& feature_match : feature_matches_lhs) {
00187     feature_indices.insert(std::make_pair(feature_match.feature_index1_,
00188                                           feature_match.feature_index2_));
00189   }
00190 
00191   // For each match in the RHS set, search for the same match in the LHS set.
00192   // If the match is not symmetric, remove it from the RHS set.
00193   auto rhs_iter = feature_matches_rhs.begin();
00194   while (rhs_iter != feature_matches_rhs.end()) {
00195     const auto& lhs_matched_iter =
00196         feature_indices.find(rhs_iter->feature_index2_);
00197 
00198     // If a symmetric match is found, keep it in the RHS set.
00199     if (lhs_matched_iter != feature_indices.end()) {
00200       if (lhs_matched_iter->second == rhs_iter->feature_index1_) {
00201         ++rhs_iter;
00202         continue;
00203       }
00204     }
00205 
00206     // Remove the non-symmetric match and continue on.
00207     feature_matches_rhs.erase(rhs_iter);
00208   }
00209 }
