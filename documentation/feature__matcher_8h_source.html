40 // This class defines a base class for feature matching. A derived class should
41 // be used to implement the specific feature matching strategy. The feature
42 // matcher is templated by a distance metric that will be used to compare
43 // distances between descriptor vectors. All matching strategies will attempt to
44 // do pairwise matches between all pairs of images. This is slow, with O(n^2) in
45 // the number of images.

139 template <typename DistanceMetric>
140 bool FeatureMatcher<DistanceMetric>::
141 MatchImages(const FeatureMatcherOptions& options,
142  PairwiseImageMatchList& image_matches) {
143  // Store the matching options locally.
144  options_ = options;
145 
146  // Iterate over all pairs of images and attempt to match them together by
147  // comparing their features.
148  for (size_t ii = 0; ii < image_features_.size(); ++ii) {
149  // Make sure this image has features.
150  if (image_features_[ii].size() == 0) {
151  continue;
152  }
153  for (size_t jj = ii+1; jj < image_features_.size(); ++jj) {
154  // Make sure this image has features.
155  if (image_features_[jj].size() == 0) {
156  continue;
157  }
158 
159  // Create an image match object and attempt to match image ii to image jj.
160  PairwiseImageMatch image_match;
161  if (!MatchImagePair(ii, jj, image_match.feature_matches_)) {
162  VLOG(1) << "Could not match image " << ii << " to image " << jj << ".";
163  continue;
164  }
165 
166  // If the image match was successful, store it.
167  image_match.image1_index_ = ii;
168  image_match.image2_index_ = jj;
169  image_matches.push_back(image_match);
170  }
171  }
172 
173  // Return whether or not we found matches between any of the images.
174  return image_matches.size() > 0;
175 }

177 template <typename DistanceMetric>
178 void FeatureMatcher<DistanceMetric>::SymmetricMatches(
179  const std::vector<LightFeatureMatch>& feature_matches_lhs,
180  std::vector<LightFeatureMatch>& feature_matches_rhs) {
181 
182  std::unordered_map<int, int> feature_indices;
183  feature_indices.reserve(feature_matches_lhs.size());
184 
185  // Add all LHS matches to the map.
186  for (const auto& feature_match : feature_matches_lhs) {
187  feature_indices.insert(std::make_pair(feature_match.feature_index1_,
188  feature_match.feature_index2_));
189  }
190 
191  // For each match in the RHS set, search for the same match in the LHS set.
192  // If the match is not symmetric, remove it from the RHS set.
193  auto rhs_iter = feature_matches_rhs.begin();
194  while (rhs_iter != feature_matches_rhs.end()) {
195  const auto& lhs_matched_iter =
196  feature_indices.find(rhs_iter->feature_index2_);
197 
198  // If a symmetric match is found, keep it in the RHS set.
199  if (lhs_matched_iter != feature_indices.end()) {
200  if (lhs_matched_iter->second == rhs_iter->feature_index1_) {
201  ++rhs_iter;
202  continue;
203  }
204  }
205 
206  // Remove the non-symmetric match and continue on.
207  feature_matches_rhs.erase(rhs_iter);
208  }
209 }
