00078 // Evaluate model on a single data element and update error.
00079 bool FundamentalMatrixRansacModel::IsGoodFit(
00080     const FeatureMatch& data_point,
00081     double error_tolerance) {
00082   const double error = EvaluateEpipolarCondition(data_point);
00083 
00084   // Test squared error against the provided tolerance.
00085   if (error * error < error_tolerance) {
00086     return true;

00091 double FundamentalMatrixRansacModel::EvaluateEpipolarCondition(
00092     const FeatureMatch& match) const {
00093   // Construct vectors for 2D keypoints in match.
00094   Eigen::Vector3d kp1, kp2;
00095   kp1 << match.feature1_.u_, match.feature1_.v_, 1;
00096   kp2 << match.feature2_.u_, match.feature2_.v_, 1;
00097 
00098   // Compute deviation from the epipolar condition.
00099   const double epipolar_condition = kp2.transpose() * F_ * kp1;
00100   return epipolar_condition;
00101 }

00117   // Make sure we don't over step.

00120     VLOG(1) << "Requested more RANSAC data samples than are available. "

00127       data_.begin(), data_.begin() + static_cast<size_t>(num_samples));

00146 // Fit a model to the provided data using the 8-point algorithm.
00147 FundamentalMatrixRansacModel FundamentalMatrixRansacProblem::FitModel(
00148     const std::vector<FeatureMatch>& input_data) const {
00149   // Create an empty fundamental matrix.
00150   Eigen::Matrix3d F;
00151 
00152   // Run the 8-point algorithm with default options.
00153   EightPointAlgorithmSolver solver;
00154   FundamentalMatrixSolverOptions options;
00155   solver.SetOptions(options);
00156 
00157   if (solver.ComputeFundamentalMatrix(input_data, F)) {
00158     // Create a new RansacModel using the computed fundamental matrix.
00159     FundamentalMatrixRansacModel model_out(F);
00160 
00161     // Record sum of squared error over all matches.
00162     model_out.error_ = 0.0;
00163     for (const auto& feature_match : input_data) {
00164       const double error = model_out.EvaluateEpipolarCondition(feature_match);
00165       model_out.error_ += error * error;
00166     }
00167 
00168     return model_out;
00169   }
00170   // Set a large error - we didn't find a model that fits the data.
00171   FundamentalMatrixRansacModel model_out(Eigen::Matrix3d::Identity());
00172   model_out.error_ = std::numeric_limits<double>::infinity();
00173   return model_out;
00174 }
