00032 // Given a path to a file (e.g. /home/erik/test.png), return the filepath up to
00033 // bu not including the last '/'.
00034 std::string GetPath(const std::string &path) {
00035   // If the input path is empty, return an empty string.
00036   if (path.empty())
00037     return std::string("");
00038 
00039   // If the input path ends in a '/', it already does not have a filename.
00040   if (path.back() == '/')
00041     return path;
00042 
00043   std::vector<std::string> path_tokens;
00044   TokenizeFileString(path, &path_tokens);
00045 
00046   // If there are no tokens and the path begins with '/', return '/'. Otherwise
00047   // the input name is a file and therefore has no path.
00048   if (path_tokens.size() == 0)
00049     return path[0] == '/' ? std::string("/") : std::string("");
00050 
00051   // Remove the last token from the tokenized path.
00052   path_tokens.erase(path_tokens.end());
00053 
00054   // Join the remaining tokens, add a prepending slash if one existed in the
00055   // input path, and return the result.
00056   std::string output_path = JoinFilepath(path_tokens);
00057   if (path[0] == '/')
00058     return PrependSlash(output_path);
00059 
00060   return output_path;
00061 }
