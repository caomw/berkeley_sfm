00160 template <typename MatrixType>
00161 void OpenCVToEigenMat(const cv::Mat& cv_mat, MatrixType& eigen_mat) {
00162   // Make sure the data is grayscale before converting to an eigen matrix.
00163   if (cv_mat.channels() != 1) {
00164     cv::Mat grayscale_mat;
00165     cv::cvtColor(cv_mat, grayscale_mat, CV_RGB2GRAY);
00166     cv::cv2eigen(grayscale_mat, eigen_mat);
00167   } else {
00168     cv::cv2eigen(cv_mat, eigen_mat);

00170 }

00172 // Non-member conversion from OpenCV to Eigen vectors.
00173 template <typename VectorType>
00174 void OpenCVToEigenVec(const cv::Mat& cv_vec, VectorType& eigen_vec) {
00175   // Make sure the data is grayscale before converting to an eigen matrix.
00176   eigen_vec.resize(cv_vec.total());
00177   if (cv_vec.channels() != 1) {
00178     cv::Mat grayscale_vec;
00179     cv::cvtColor(cv_vec, grayscale_vec, CV_RGB2GRAY);
00180     for (size_t ii = 0; ii < grayscale_vec.total(); ++ii)
00181       eigen_vec(ii) = grayscale_vec.at<float>(ii);
00182   } else {
00183     for (size_t ii = 0; ii < cv_vec.total(); ++ii)
00184       eigen_vec(ii) = cv_vec.at<float>(ii);
00185   }
00186 }
