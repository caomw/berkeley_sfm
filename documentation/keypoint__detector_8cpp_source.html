00148 bool KeypointDetector::DetectKeypoints(const Image& image,
00149                                        KeypointList& keypoints_out) {
00150   // Make the user has called SetDetector().
00151   if (detector_type_.empty()) {
00152     VLOG(1) << "Detector has not been specified via SetDetector(). Failed to "
00153                "detect keypoints.";
00154     return false;
00155   }
00156   CHECK(detector_) << "The feature detector is null.";
00157 
00158   // Clear the output.
00159   keypoints_out.clear();
00160 
00161   // Convert the input image to OpenCV's format. Note that features must be
00162   // detected on the grayscale image, and that the image format must be CV_8U.
00163   cv::Mat cv_image;
00164   image.ToCV(cv_image);
00165   cv_image.convertTo(cv_image, CV_8U, 255);
00166 
00167   // Detect keypoints in the image.
00168   try {
00169     detector_->detect(cv_image, keypoints_out);
00170   } catch (const std::exception& e) {
00171     VLOG(1) << "Failed to detect keypoints: " << e.what();
00172     return false;
00173   }
00174 
00175   return true;
00176 }
