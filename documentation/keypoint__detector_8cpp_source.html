48 bool KeypointDetector::SetDetector(const std::string& detector_type) {
49  // Set the detector type.
50  detector_type_ = detector_type;
51  bool valid_detector_type = true;
52 
53  // Create an OpenCV feature detector.
54  if (detector_type.compare("SIFT") == 0) {
55  detector_ = cv::FeatureDetector::create("SIFT");
56  } else if (detector_type.compare("SURF") == 0) {
57  detector_ = cv::FeatureDetector::create("SURF");
58  } else if (detector_type.compare("FAST") == 0) {
59  detector_ = cv::FeatureDetector::create("FAST");
60  } else if (detector_type.compare("STAR") == 0) {
61  detector_ = cv::FeatureDetector::create("STAR");
62  } else if (detector_type.compare("ORB") == 0) {
63  detector_ = cv::FeatureDetector::create("ORB");
64  } else if (detector_type.compare("BRISK") == 0) {
65  detector_ = cv::FeatureDetector::create("BRISK");
66  } else if (detector_type.compare("MSER") == 0) {
67  detector_ = cv::FeatureDetector::create("MSER");
68  } else if (detector_type.compare("GFTT") == 0) {
69  detector_ = cv::FeatureDetector::create("GFTT");
70  } else if (detector_type.compare("HARRIS") == 0) {
71  detector_ = cv::FeatureDetector::create("HARRIS");
72  } else if (detector_type.compare("DENSE") == 0) {
73  detector_ = cv::FeatureDetector::create("Dense");
74  } else if (detector_type.compare("SIMPLEBLOB") == 0) {
75  detector_ = cv::FeatureDetector::create("SimpleBlob");
76  } else {
77  VLOG(1) << "Detector type \"" << detector_type
78  << "\"is not available. Defaulting to FAST.";
79  detector_ = cv::FeatureDetector::create("FAST");
80  valid_detector_type = false;
81  detector_type_ = "FAST";
82  }
83 
84  return valid_detector_type;
85 }

89  // Make the user has called SetDetector().
90  if (detector_type_.empty()) {
91  VLOG(1) << "Detector has not been specified via SetDetector(). Failed to "

93  return false;
94  }
95 
96  CHECK(detector_) << "The feature detector is null.";
97 
98  // Convert the input image to OpenCV's format. Note that features must be
99  // detected on the grayscale image, and that the image format must be CV_8U.
100  cv::Mat cv_image;
101  image.ToCV(cv_image);
102  cv_image.convertTo(cv_image, CV_8U, 255);
103 
104  // Detect keypoints in the image.
105  try {
106  detector_->detect(cv_image, keypoints_out);
107  } catch (const std::exception& e) {
108  VLOG(1) << "Failed to detect keypoints: " << e.what();
109  return false;
110  }
111 
112  return true;
113 }
