00090 template <typename DistanceMetric>
00091 bool NaiveFeatureMatcher<DistanceMetric>::MatchImagePair(
00092     int image_index1,
00093     int image_index2,
00094     std::vector<FeatureMatch>& feature_matches) {
00095   feature_matches.clear();
00096 
00097   // Get the descriptors corresponding to these two images.
00098   std::vector<Descriptor>& descriptors1 = this->image_descriptors_[image_index1];
00099   std::vector<Descriptor>& descriptors2 = this->image_descriptors_[image_index2];
00100 
00101   // Normalize descriptors if required.
00102   if (DistanceMetric::RequiresNormalizedDescriptors()) {
00103     for (auto& descriptor : descriptors1) {
00104       descriptor.normalize();
00105     }
00106     for (auto& descriptor : descriptors1) {
00107       descriptor.normalize();
00108     }
00109   }
00110 
00111   // Compute forward (and reverse, if applicable) matches.
00112   LightFeatureMatchList light_feature_matches;
00113   ComputePutativeMatches(descriptors1, descriptors2, light_feature_matches);
00114 
00115   // Check that we got enough matches here. If we didn't, reverse matches won't
00116   // help us.
00117   if (light_feature_matches.size() < this->options_.min_num_feature_matches) {
00118     return false;
00119   }
00120 
00121   if (this->options_.require_symmetric_matches) {
00122     LightFeatureMatchList reverse_light_feature_matches;
00123     ComputePutativeMatches(descriptors2, descriptors1,
00124                            reverse_light_feature_matches);
00125     this->SymmetricMatches(reverse_light_feature_matches,
00126                            light_feature_matches);
00127   }
00128 
00129   if (light_feature_matches.size() < this->options_.min_num_feature_matches) {
00130     return false;
00131   }
00132 
00133   // Convert from LightFeatureMatchList to FeatureMatchList for the output.
00134   // Return matches in sorted order.
00135   std::sort(light_feature_matches.begin(), light_feature_matches.end(),
00136             LightFeatureMatch::SortByDistance);
00137 
00138   // Check how many features the user wants.
00139   unsigned int num_features_out = light_feature_matches.size();
00140   if (this->options_.only_keep_best_matches) {
00141     num_features_out =
00142         std::min(num_features_out, this->options_.num_best_matches);
00143   }
00144 
00145   for (int ii = 0; ii < num_features_out; ++ii) {
00146     const auto& match = light_feature_matches[ii];
00147 
00148     const Feature& matched_feature1 =
00149         this->image_features_[image_index1][match.feature_index1_];
00150     const Feature& matched_feature2 =
00151         this->image_features_[image_index2][match.feature_index2_];
00152 
00153     feature_matches.emplace_back(matched_feature1, matched_feature2);
00154   }
00155 
00156   return true;
00157 }

00159 template <typename DistanceMetric>
00160 void NaiveFeatureMatcher<DistanceMetric>::ComputePutativeMatches(
00161     const std::vector<Descriptor>& descriptors1,
00162     const std::vector<Descriptor>& descriptors2,
00163     std::vector<LightFeatureMatch>& putative_matches) {
00164   putative_matches.clear();
00165 
00166   // Create a distance metric for descriptor comparison.
00167   DistanceMetric distance;
00168 
00169   // Store all matches and their distances.
00170   for (size_t ii = 0; ii < descriptors1.size(); ++ii) {
00171     LightFeatureMatchList one_way_matches;
00172     for (size_t jj = 0; jj < descriptors2.size(); ++jj) {
00173       double dist = distance(descriptors1[ii], descriptors2[jj]);
00174       one_way_matches.emplace_back(ii, jj, dist);
00175     }
00176 
00177     // Store the best match for this element of features2.
00178     if (this->options_.use_lowes_ratio) {
00179       // Sort by distance. We only care about the distances between the best 2
00180       // matches for the Lowes ratio test.
00181       std::partial_sort(one_way_matches.begin(),
00182                         one_way_matches.begin() + 1,
00183                         one_way_matches.end(),
00184                         LightFeatureMatch::SortByDistance);
00185 
00186       // The second best match must be within the lowes ratio of the best match.
00187       double lowes_ratio_squared =
00188           this->options_.lowes_ratio * this->options_.lowes_ratio;
00189       if (one_way_matches[0].distance_ <
00190           lowes_ratio_squared * one_way_matches[1].distance_) {
00191         putative_matches.emplace_back(one_way_matches[0]);
00192       }
00193     } else {
00194       putative_matches.emplace_back(one_way_matches[0]);
00195     }
00196   }
00197 }
