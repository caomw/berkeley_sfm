120 // Convert to axis-angle representation.
121 Eigen::VectorXd Pose::ToAxisAngle() {
122  // From https://en.wikipedia.org/wiki/Axis-angle-representation.
123  const double angle = acos(0.5 * (Rt_.trace() - 2.0));
124  Eigen::Vector3d axis = Eigen::Vector3d(Rt_(2, 1) - Rt_(1, 2),
125  Rt_(0, 2) - Rt_(2, 0),
126  Rt_(1, 0) - Rt_(0, 1)) * 0.5 / sin(angle);
127 
128  axis /= axis.norm();
129  return axis * angle;
130 }

132 // Convert to matrix representation.
133 Eigen::Matrix4d Pose::FromAxisAngle(const Eigen::Vector3d& aa) {
134  // From https://en.wikipedia.org/wiki/Rotation_matrix.
135  double angle = aa.norm();
136  Eigen::Vector3d axis = aa / angle;
137 
138  Eigen::Matrix3d cross;
139  cross <<
140  0.0, -axis(2), axis(1),
141  axis(2), 0.0, -axis(0),
142  -axis(1), axis(0), 0.0;
143 
144  Eigen::Matrix3d tensor;
145  tensor <<
146  axis(0)*axis(0), axis(0)*axis(1), axis(0)*axis(2),
147  axis(0)*axis(1), axis(1)*axis(1), axis(1)*axis(2),
148  axis(0)*axis(2), axis(1)*axis(2), axis(2)*axis(2);
149 
150  Eigen::Matrix3d R =
151  cos(angle) * Eigen::Matrix3d::Identity() +
152  sin(angle) * cross + (1-cos(angle)) * tensor;
153  Rt_.block(0, 0, 3, 3) = R;
154 
155  return Rt_;
156 }
