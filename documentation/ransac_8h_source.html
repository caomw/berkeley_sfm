00082 template <typename DataType, typename ModelType>
00083 bool Ransac<DataType, ModelType>::Run(
00084     RansacProblem<DataType, ModelType>& problem) const {
00085   // By default, a valid model has not been found.
00086   problem.SetSolutionFound(false);
00087 
00088   // Set the initial error to something very large.
00089   double best_error = std::numeric_limits<double>::infinity();
00090 
00091   // Proceed for options_.iterations iterations of RANSAC.
00092   for (unsigned int iter = 0; iter < options_.iterations; ++iter) {
00093     // Sample data points.
00094     std::vector<DataType> sampled = problem.SampleData(options_.num_samples);
00095 
00096     // Fit a model to the sampled data points.
00097     ModelType initial_model = problem.FitModel(sampled);
00098 
00099     // Which of the remaining points are also inliers under this model?
00100     std::vector<DataType> unsampled = problem.RemainingData(options_.num_samples);
00101     std::vector<DataType> inliers(sampled);
00102     for (const auto& not_sampled_data_point : unsampled) {
00103       if (initial_model.IsGoodFit(not_sampled_data_point,
00104                                   options_.acceptable_error)) {
00105         inliers.push_back(not_sampled_data_point);
00106       }
00107     }
00108 
00109     // Check if we have enough inliers to consider this a good model.
00110     if (inliers.size() >= options_.minimum_num_inliers) {
00111 
00112       // Test how good this model is.
00113       ModelType better_model = problem.FitModel(inliers);
00114 
00115       // Is this the best model yet?
00116       double this_error = better_model.Error();
00117       if (this_error < best_error) {
00118         best_error = this_error;
00119         problem.SetModel(better_model);
00120         problem.SetInliers(inliers);
00121         problem.SetSolutionFound(true);
00122       }
00123     }
00124   }
00125 
00126   // See if RANSAC found a solution.
00127   if (!problem.SolutionFound()) {
00128     VLOG(1) << "RANSAC failed to find a solution in " << options_.iterations
00129             << " iterations.";
00130     return false;
00131   }
00132 
00133   return true;
00134 }
